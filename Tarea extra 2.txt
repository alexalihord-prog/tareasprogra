function jugar() {
let continuar = true;
while (continuar) {
const numeroComputadora = Math.floor(Math.random() * 9) + 1;
let numeroUsuario;
do 
{
numeroUsuario = parseInt(prompt("Ingrese un número entre 3 y 6:"));
if (numeroUsuario < 3 || numeroUsuario > 6 || isNaN(numeroUsuario)) {
alert("Error: El número debe estar exclusivamente entre 3 y 6.");
}
} 
while (numeroUsuario < 3 || numeroUsuario > 6 || isNaN(numeroUsuario));
let prediccion = prompt(`¿Tu número (${numeroUsuario}) es MAYOR, MENOR o IGUAL al de la computadora?`).toLowerCase();
let resultadoReal;
if (numeroUsuario > numeroComputadora) {
resultadoReal = "mayor";
} 
else if (numeroUsuario < numeroComputadora) 
{
resultadoReal = "menor";
} 
else 
{
resultadoReal = "igual";
}
let mensajeResultado = `La computadora eligió ${numeroComputadora}, usted eligió ${numeroUsuario}. `;
if (prediccion === resultadoReal) {
alert(mensajeResultado + `Su elección es ${prediccion}. ¡Ha adivinado!`);
} else {
alert(mensajeResultado + `Su elección fue ${prediccion}, pero era ${resultadoReal}. No ha adivinado.`);
}
let respuesta = prompt("¿Quiere jugar otra vez? (SI/NO)").toUpperCase();
if (respuesta !== "SI") {
 continuar = false;
alert("Juego terminado.\Nombre: Sergio Garcia\nCarnet:26000866 ");
}
}
}
jugar();
